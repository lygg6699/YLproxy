```powershell
$cpu = (Get-WmiObject -Class Win32_Processor).NumberOfCores

Test-NetConnection -ComputerName 8.8.8.8 -CommonTCPPort DNS
```

## 部署流程（详细步骤）

### 阶段 1：准备部署包（第 1 天）

**步骤 1.1：编译发布版本**

```powershell
# 进入项目目录
cd "e:\GZQ\YLXCX\YLproxy"

# 清理旧构建
dotnet clean
Write-Host "清理完成"

# 还原 NuGet 包
dotnet restore
if ($LASTEXITCODE -ne 0) {
    Write-Error "包还原失败"
    exit 1
}
Write-Host "包还原完成"

# 编译发布版本
dotnet publish -c Release -o ".\publish" --no-restore
if ($LASTEXITCODE -ne 0) {
    Write-Error "发布失败"
    exit 1
}
Write-Host "发布完成"

# 验证文件完整性
$publishFiles = Get-ChildItem ".\publish" -Recurse | Measure-Object
Write-Host "发布文件数: $($publishFiles.Count)"
```

**步骤 1.2：创建部署包**

```powershell
# 打包为 ZIP
$packageDate = Get-Date -Format "yyyyMMdd_HHmmss"
$packageName = "YLproxy_Release_$packageDate.zip"

Compress-Archive -Path ".\publish\*" `
                 -DestinationPath "..\$packageName" `
                 -Force

Write-Host "部署包已创建: $packageName"
Write-Host "大小: $(([math]::Round((Get-Item "..\$packageName").Length / 1MB, 2))) MB"

# 验证包完整性
$zip = Add-Type -AssemblyName System.IO.Compression
$file = [System.IO.Compression.ZipFile]::OpenRead("..\$packageName")
Write-Host "包内文件数: $($file.Entries.Count)"
$file.Dispose()
```

**步骤 1.3：生成部署检查单**

```powershell
# 生成部署清单
@{
    Version = "1.0.0"
    BuildDate = (Get-Date)
    Package = $packageName
    Files = @(
        "YLproxy.exe"
        "YLproxy.dll"
        "YLproxy.Core.dll"
        "YLproxy.GUI.dll"
        "YLproxy.Infrastructure.dll"
        "YLproxy.Models.dll"
        "YLproxy.Proxy.dll"
        "YLproxy.Utils.dll"
    )
} | ConvertTo-Json | Out-File "deployment-manifest.json"

Write-Host "部署清单已生成"
```

### 阶段 2：部署到目标服务器（第 2 天）

**步骤 2.1：传输部署包**

```powershell
# 方式 A：本地部署（测试环境）
$targetServer = "."
$deployPath = "C:\YLproxy"

# 方式 B：远程部署（生产环境）
$targetServer = "prod-server.example.com"
$deployPath = "C:\YLproxy"
$session = New-PSSession -ComputerName $targetServer

# 1. 创建部署目录
Invoke-Command -Session $session -ScriptBlock {
    $path = "C:\YLproxy"
    if (-not (Test-Path $path)) {
        New-Item -ItemType Directory -Path $path -Force | Out-Null
        Write-Host "创建目录: $path"
    }
}

# 2. 复制文件
Copy-Item ".\publish\*" `
          -Destination "$deployPath\" `
          -Recurse -Force `
          -ToSession $session

Write-Host "文件已复制到目标服务器"
```

**步骤 2.2：验证文件完整性**

```powershell
# 验证关键文件存在
$expectedFiles = @(
    "YLproxy.exe"
    "YLproxy.dll"
    "data\config.json"
    "runtime\3proxy\bin64\3proxy.exe"
)

Invoke-Command -Session $session -ScriptBlock {
    foreach ($file in $using:expectedFiles) {
        $path = "C:\YLproxy\$file"
        if (Test-Path $path) {
            Write-Host "✓ $file"
        } else {
            Write-Host "✗ $file (缺失)"
        }
    }
}
```

**步骤 2.3：停止旧应用实例**

```powershell
# 如果旧版本正在运行，先停止
Invoke-Command -Session $session -ScriptBlock {
    # 查找 YLproxy 进程
    $proc = Get-Process -Name "YLproxy" -ErrorAction SilentlyContinue
    if ($proc) {
        Write-Host "停止现有 YLproxy 进程..."
        Stop-Process -InputObject $proc -Force
        Start-Sleep -Seconds 2
        Write-Host "进程已停止"
    }
}
```

**步骤 2.4：配置文件迁移**

```powershell
# 如果从旧版本升级，迁移旧配置
Invoke-Command -Session $session -ScriptBlock {
    $newPath = "C:\YLproxy"
    $oldPath = "C:\OldYLproxy"  # 如果存在旧版本
    
    if (Test-Path "$oldPath\data\config.json") {
        Write-Host "检测到旧配置，进行迁移..."
        
        # 1. 备份新配置（如果存在）
        if (Test-Path "$newPath\data\config.json") {
            Copy-Item "$newPath\data\config.json" `
                      "$newPath\data\config.json.new"
        }
        
        # 2. 迁移旧配置
        Copy-Item "$oldPath\data\config.json" `
                  "$newPath\data\config.json" `
                  -Force
        
        Write-Host "旧配置已迁移"
    }
}
```

### 阶段 3：启动和初始化（第 2 天）

**步骤 3.1：启动应用**

```powershell

    & ".\YLproxy.exe" 2>&1 | Write-Host

    # 验证启动

    if ($proc) {
        Write-Host "✓ YLproxy 已启动 (PID: $($proc.Id))"
    } else {
        Write-Host "✗ YLproxy 启动失败"
    }
}
```

**步骤 3.2：验证配置加载**

```powershell
Invoke-Command -Session $session -ScriptBlock {
    # 等待配置加载
    Start-Sleep -Seconds 3
    
    # 检查配置文件
    $configPath = "C:\YLproxy\data\config.json"
    if (Test-Path $configPath) {
        $config = Get-Content $configPath | ConvertFrom-Json
        Write-Host "✓ 配置已加载"
        Write-Host "  代理数量: $($config.proxies.Count)"

    # 检查日志
    $logPath = "C:\YLproxy\logs\latest.log"
    if (Test-Path $logPath) {
        $errors = @(Get-Content $logPath | Select-String "ERROR")
        if ($errors.Count -eq 0) {
            Write-Host "✓ 日志正常"
        } else {
            Write-Host "⚠️ 日志中发现错误:"
            $errors | Select-Object -First 5
        }
    }
}
```

## 性能基准测试

### 基准测试项目

**基准线定义**

| 指标 | 预期值 | 警告阈值 | 错误阈值 |
|---|---|---|---|
| CPU 使用率 | < 20% | 50% | > 70% |
| 内存占用 | < 300 MB | 500 MB | > 800 MB |
| 响应延迟 | < 50 ms | 100 ms | > 200 ms |
| 并发连接 | 支持 100+ | < 50 | 失败 |
| 吞吐量 | > 1 Mbps | 降低 50% | < 500 Kbps |

**测试脚本**

```powershell
# 性能基准测试脚本
param(
    [string]$ProxyAddress = "127.0.0.1",
    [int]$ProxyPort = 9999,
    [int]$TestDuration = 60,
    [int]$ConcurrentConnections = 10
)

$startTime = Get-Date
$endTime = $startTime.AddSeconds($TestDuration)

# 获取基准线数据
Write-Host "性能基准测试开始..."
Write-Host "代理: $ProxyAddress:$ProxyPort"
Write-Host "持续时间: $TestDuration 秒"
Write-Host "并发连接: $ConcurrentConnections"

# 记录初始进程状态
$proc = Get-Process -Name "YLproxy"
$initialCPU = $proc.CPU
$initialMemory = $proc.WorkingSet

Write-Host ""
Write-Host "初始状态:"
Write-Host "  CPU: $initialCPU%"
Write-Host "  内存: $([math]::Round($initialMemory / 1MB, 2)) MB"

# 模拟测试（实际需要真实代理测试）
Start-Sleep -Seconds $TestDuration

# 记录最终进程状态
$proc = Get-Process -Name "YLproxy"
$finalCPU = $proc.CPU
$finalMemory = $proc.WorkingSet

Write-Host ""
Write-Host "最终状态:"
Write-Host "  CPU: $finalCPU%"
Write-Host "  内存: $([math]::Round($finalMemory / 1MB, 2)) MB"
Write-Host "  内存增长: $([math]::Round(($finalMemory - $initialMemory) / 1MB, 2)) MB"

# 验证
Write-Host ""
Write-Host "基准测试结果:"
if ($finalCPU -lt 50) {
    Write-Host "  ✓ CPU 使用率正常"
} else {
    Write-Host "  ⚠️ CPU 使用率偏高"
}

if ($finalMemory -lt 800MB) {
    Write-Host "  ✓ 内存占用正常"
} else {
    Write-Host "  ⚠️ 内存占用偏高"
}
```

## 回滚策略

### 快速回滚流程

**回滚触发条件**
```
🔴 回滚条件（任一满足即回滚）：
- 应用无法启动
- 关键功能失效
- 性能恶化超过 50%
- CPU 持续 > 90%
- 内存持续 > 1 GB
- 代理连通性测试全部失败
- 数据丢失或损坏
```

**回滚流程（5 分钟内完成）**

```powershell
# 第 1 步：停止现有版本（30 秒）
Stop-Process -Name "YLproxy" -Force
Write-Host "当前版本已停止"

# 第 2 步：恢复旧版本（1 分钟）
$backupPath = "C:\YLproxy.backup"  # 升级前备份目录
Copy-Item "$backupPath\*" "C:\YLproxy\" -Recurse -Force
Write-Host "旧版本已恢复"

# 第 3 步：恢复配置（30 秒）
$configBackup = Get-ChildItem "C:\YLproxy\backup" -Filter "config.json.backup.*" | 
                Sort-Object LastWriteTime -Descending | Select-Object -First 1
if ($configBackup) {
    Copy-Item $configBackup.FullName "C:\YLproxy\data\config.json" -Force
    Write-Host "配置已恢复: $($configBackup.Name)"
}

# 第 4 步：重启应用（2 分钟）
cd "C:\YLproxy"
& ".\YLproxy.exe"
Start-Sleep -Seconds 10

# 第 5 步：验证（1 分钟）
$proc = Get-Process -Name "YLproxy" -ErrorAction SilentlyContinue
if ($proc) {
    Write-Host "✓ 应用已恢复运行"
    Write-Host "回滚完成，总耗时：约 5 分钟"
} else {
    Write-Host "✗ 应用启动失败，需要手动干预"
}
```

### 完全回滚（数据恢复）

```powershell
# 如果简单回滚不成功，进行完全恢复

# 第 1 步：清空新版本
Remove-Item "C:\YLproxy" -Recurse -Force

    if (Test-Path "C:\YLproxy\$file") {
```

## 灾难恢复

### 恢复场景 1：配置文件损坏

**症状**：应用无法启动，报错"配置文件格式错误"

**恢复步骤**

```powershell
# 1. 停止应用
Stop-Process -Name "YLproxy" -Force

# 2. 查看备份列表
Get-ChildItem "C:\YLproxy\backup" -Filter "config.json.backup.*"

# 3. 选择最近的有效备份恢复
$latestBackup = Get-ChildItem "C:\YLproxy\backup" -Filter "config.json.backup.*" |
                Sort-Object LastWriteTime -Descending |
                Select-Object -First 1

Copy-Item $latestBackup.FullName "C:\YLproxy\data\config.json" -Force

# 4. 验证配置格式
$config = Get-Content "C:\YLproxy\data\config.json" | ConvertFrom-Json -ErrorAction Stop
Write-Host "配置文件有效"

# 5. 重启应用
cd "C:\YLproxy"
& ".\YLproxy.exe"
```

### 恢复场景 2：磁盘空间不足

**症状**：日志写入失败，应用性能下降

**恢复步骤**

```powershell
# 1. 检查磁盘空间
$disk = Get-PSDrive C
Write-Host "可用空间: $([math]::Round($disk.Free / 1GB, 2)) GB"

# 2. 清理日志归档
Get-ChildItem "C:\YLproxy\logs\archive" -Filter "*.log" |
    Where-Object { $_.LastWriteTime -lt (Get-Date).AddDays(-30) } |
    Remove-Item -Force
Write-Host "过期日志已清理"

# 3. 压缩旧日志
Get-ChildItem "C:\YLproxy\logs\archive" -Filter "*.log" |
    Where-Object { $_.LastWriteTime -lt (Get-Date).AddDays(-7) } |
    ForEach-Object {
        Compress-Archive $_.FullName "$($_.FullName).zip" -Force
        Remove-Item $_.FullName -Force
    }
Write-Host "旧日志已压缩"

# 4. 检查临时文件
Remove-Item "C:\YLproxy\temp\*" -Force -ErrorAction SilentlyContinue

# 5. 重新启动应用
cd "C:\YLproxy"
& ".\YLproxy.exe"
```

## 常见部署问题

### 问题 1：部署失败，报错"目录无法写入"

**原因**
- 目录权限不足
- 目录被其他进程锁定

**解决方案**

```powershell
# 1. 检查目录权限
icacls "C:\YLproxy"

# 2. 赋予权限
icacls "C:\YLproxy" /grant:r "NT AUTHORITY\NETWORK SERVICE:(OI)(CI)F"

# 3. 关闭所有 YLproxy 进程
Stop-Process -Name "YLproxy" -Force

# 4. 重新尝试部署
```

### 问题 2：回滚失败，新旧版本共存

**原因**
- 备份路径错误
- 权限问题导致覆盖失败

**解决方案**

```powershell
# 1. 手动清理新版本
Remove-Item "C:\YLproxy\*.exe" -Force

# 2. 手动复制旧版本
Copy-Item "C:\YLproxy.backup\*" "C:\YLproxy\" -Recurse -Force

# 3. 重启应用
cd "C:\YLproxy"
& ".\YLproxy.exe"
```

### 问题 3：部署后配置丢失

**原因**
- 配置文件路径变更
- 备份恢复时出错

**解决方案**

```powershell
# 1. 查找配置文件备份
Get-ChildItem "C:\YLproxy\backup" -Recurse -Filter "config.json*"

# 2. 恢复配置
$backup = Get-ChildItem "C:\YLproxy\backup" -Filter "config.json.backup.*" |
          Sort-Object LastWriteTime -Descending |
          Select-Object -First 1
          
Copy-Item $backup.FullName "C:\YLproxy\data\config.json" -Force

# 3. 验证配置
$config = Get-Content "C:\YLproxy\data\config.json" | ConvertFrom-Json
Write-Host "配置项数: $($config.proxies.Count)"

# 4. 重启应用
```
